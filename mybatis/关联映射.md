~~~xml
<select id="getPeopleByPassportId" parameterType="int"
		resultType="People">
		select * from
		People
		where passportId=#{id};
	</select>

	<resultMap type="People" id="GetPeople">
		<id column="id" property="id"></id>
		<result column="serialno" property="serialno" />
		<result column="expire" property="expire" />
		<association property="people" column="id"
			select="getPeopleByPassportId"></association>
        #另外使用
	</resultMap>


	<select id="findById" parameterType="int" resultMap='GetPeople'>
		select *
		from Passport
		where id=#{id};
	</select>

	<select id="findAll">
		select *
		from Passport;
	</select>
~~~
